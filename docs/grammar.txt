IGNORE : ' \t\x0c'
DELIM : r';'
ICONST : r'\d+'
ID : r'[A-Za-z_][\w_]*'
NEWLINE : r'\n+'
EQUAL : r'='
LPAREN = r'\('
RPAREN = r'\)'
COMMA = r','

program : program statement DELIM
        | statement DELIM

statement : declare_statement
          | assign_statement
          | init_statement
          | command_statement

declare_statement : type_specifier ID

init_statement : INICIAR LPAREN ID RPAREN

assign_statement : ID EQUAL expression

type_specifier : INT
               | COLA

expression : constant

constant : ICONST

command_statement : command LPAREN command_args RPAREN

command : ACOLAR
        | DESACOLAR
        | PRIMERO
        | APILAR
        | DESAPILAR
        | TOPE
        | INGRESAR

command_args : ID COMMA ID
             | ID COMMA constant
