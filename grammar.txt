IGNORE : ' \t\x0c'
DELIM : r';'
ICONST : r'\d+'
ID : r'[A-Za-z_][\w_]*'
NEWLINE : r'\n+'
EQUAL : r'='
LPAREN = r'\('
RPAREN = r'\)'

program : program statement
        | statement

statement : declare_statement DELIM
          | init_statement DELIM

declare_statement : type_specifier ID

init_statement : assign_statement
               | INICIAR LPAREN ID RPAREN

assign_statement : ID EQUAL constant

type_specifier : INT
               | COLA

constant : ICONST
